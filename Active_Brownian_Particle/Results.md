$$
\begin{align}
\vec{I}[\rho](\vec{r},\phi,t)&= \int rdrd\psi_R d\phi^\prime U^\prime(r)\hat{u}(\psi_R)g\big(r,\psi_R-\phi,\phi^\prime-\phi\big)\rho(\vec{r} + r\hat{u}(\psi_R),\phi^\prime,t)\\
&=\sum_{n, m, n_1, n_2} D_{nm} X_{n_2} Y_{C,A} \sin(n_2\phi + \psi_{n_2}+\eta_{C,A} )
\end{align}
$$
Where
$$X_{n_2}=|\rho_{n_2}|$$
$$\psi_{n_2}=\arg(\rho_{n_2})$$
$$Y_{C,A}=\sqrt{C_R^2A_s^2+C_I^2A_a^2}$$
$$\eta_{C,A}=\tan^{-1}\frac{C_IA_a}{C_RA_s}$$
$$
\begin{align}
C(n,m,n_1)
&=\pi\int_0^{2\pi}\cos^n\theta\sin^m\theta\ e^{in_1\theta}d\theta\\
&=C_R+iC_I,\ C_R, C_I\in\mathbb{R}
\end{align}
$$
$$A_{s/a}(n, m, n_1, n_2)=\int_0^\infty U^\prime(r)B_{s/a}(n_1, n_2, r) r^{n+m+1}dr$$
$$g(r, \theta_1,\theta_2)=\sum_{n_1,n_2}B_s\cos n_1\theta_1\cos n_2\theta_2+B_a\sin n_1\theta_1\sin n_2\theta_2$$
$$\theta_1=\psi_R-\phi$$
$$\theta_2=\phi^\prime-\phi$$
